289 LUResult L_U() const {
290 // Check for square matrix
291 if(this->rows != this->cols) {
292 throw std::invalid_argument("Not square");
293 }
294
295 Matrix L = eye(this->rows);
296 Matrix U = Zeros(this->rows, this->cols);
297 std::vector<int> P(this->rows);
298 for(int i = 0; i < this->rows; i++) {
299 P[i] = i;
300 }
301
302 Matrix A_work(*this);
303
304 for(int j = 0; j < this->cols; j++) {
305 int pivot_row = j;
306 double max_val = std::abs(A_work.MyMAT[j][j]);
307
308 for(int i = j+1; i < this->rows; i++) {
309 if(std::abs(A_work.MyMAT[i][j]) > max_val) {
310 pivot_row = i;
311 max_val = std::abs(A_work.MyMAT[i][j]);
312 }
313 }
314
315 if(pivot_row != j) {
316 for(int k = 0; k < this->cols; k++) {
317 std::swap(A_work.MyMAT[j][k], A_work.MyMAT[pivot_row][k]);
318 }
319
320 std::swap(P[j], P[pivot_row]);
321
322 if(j > 0) {
323 for(int k = 0; k < j; k++) {
324 std::swap(L.MyMAT[j][k], L.MyMAT[pivot_row][k]);
325 }
326 }
327 }
328
329 if(std::abs(A_work.MyMAT[j][j]) < 1e-10) {
330 throw std::runtime_error("Matrix is singular or nearly singular");
331 }
332
333 for(int i = 0; i <= j; i++) {
334 double sum = 0.0;
335 for(int k = 0; k < i; k++) {
336 sum += L.MyMAT[i][k] * U.MyMAT[k][j];
337 }
338 U.MyMAT[i][j] = A_work.MyMAT[i][j] - sum;
339 }
340
341 for(int i = j+1; i < this->rows; i++) {
342 double sum = 0.0;
343 for(int k = 0; k < j; k++) {
344 sum += L.MyMAT[i][k] * U.MyMAT[k][j];
345 }
346 L.MyMAT[i][j] = (A_work.MyMAT[i][j] - sum) / U.MyMAT[j][j];
347 }
348 }
349
350 LUResult result;
351 result.L = L.MyMAT;
352 result.U = U.MyMAT;
353 result.P = P;
354 return result;
355 }

392 QRresult QR_fact() const {
393 Matrix Q = eye(this->rows);
394 Matrix Aupdate = *this;
395
396 for (int i = 0; i < std::min(this->rows-1, this->cols); i++) {
397 My_Vec vecx(this->rows - i);
398 for (int row = i; row < this->rows; row++) {
399 vecx.myvector[row - i] = Aupdate.MyMAT[row][i];
400 }
401
402 double n = vecx.Norm();
403
404 My_Vec unit = My_Vec::unit_vec(0, this->rows - i); // Use consistent naming
405
406 My_Vec reflec_vec;
407 if (vecx.myvector[0] < 0) {
408 reflec_vec = vecx + unit.Scalar_Mul(n);
409 } else {
410 reflec_vec = vecx - unit.Scalar_Mul(n);
411 }
412
413 double normalize_ref = reflec_vec.Norm();
414
415 // Check for zero vector (avoid division by zero)
416 if (normalize_ref < 1e-10) {
417 continue; // Skip this iteration if vector is too small
418 }
419
420 My_Vec V = reflec_vec.Scalar_Mul(1.0 / normalize_ref);
421
422 // Create Householder matrix H = I - 2*vv^T
423 Matrix I = Matrix::eye(V.length);
424 Matrix vvT = Matrix::Outer_Product(V, V);
425 Matrix Householder = I - vvT.Scalar_Mul(2.0);
426
427 // Embed the Householder matrix into a larger identity matrix
428 Matrix Hprime = Matrix::eye(this->rows);
429 for (int r = 0; r < Householder.rows; r++) {
430 for (int c = 0; c < Householder.cols; c++) {
431 Hprime.MyMAT[i + r][i + c] = Householder.MyMAT[r][c];
432 }
433 }
434
435 Aupdate = Hprime * Aupdate;
436 Q = Q * Hprime;
437 }
