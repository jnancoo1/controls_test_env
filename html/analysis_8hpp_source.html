326 std::tuple<Eigen::MatrixXd, Eigen::MatrixXd> controllability_decomposition(const Discrete_StateSpace_System& System){
327 Eigen::MatrixXd cont_mat=compute_controllability_matrix(System);
328 int n=cont_mat.rows();
329 Eigen::ColPivHouseholderQR <Eigen::MatrixXd> QR(cont_mat);
330 Eigen::MatrixXd Q=QR.householderQ();
331 int r=QR.rank();
332 Eigen::MatrixXd T(n, n);
333 T << Q.leftCols(r), Q.rightCols(n - r);
334
335 Eigen::MatrixXd Aprime=(T.inverse())*System.A*T;
336 Eigen::MatrixXd Bprime=(T.inverse())*System.B;
337 Eigen::MatrixXd Cprime=System.C*T;
338
339 Eigen::MatrixXd A_cc=Aprime.topLeftCorner(r,r);
340 Eigen::MatrixXd A_cu=Aprime.topRightCorner(r,n-r);
341 std::tuple<Eigen::MatrixXd, Eigen::MatrixXd> Ans={A_cc,A_cu};
342
343 return Ans;

395 Eigen::MatrixXd compute_controllability_gramian(const Discrete_StateSpace_System& System) {
396
397 Eigen::MatrixXd Q = System.B * System.B.transpose();
398 int n = System.A.rows();
399 Eigen::MatrixXd I = Eigen::MatrixXd::Identity(n, n);
400 Eigen::VectorXd vecQ = Eigen::Map<const Eigen::VectorXd>(Q.data(), Q.size());
401
402 Eigen::MatrixXd kron1 = Eigen::kroneckerProduct(System.A, I);
403 Eigen::MatrixXd kron2 = Eigen::kroneckerProduct(I, System.A);
404
405 Eigen::VectorXd w = (kron1 + kron2).fullPivLu().solve(-vecQ);
406
407 Eigen::MatrixXd W = Eigen::Map<Eigen::MatrixXd>(w.data(), n, n);

465 std::vector<std::complex<double>> generate_z_grid(
466 double r_min, double r_max,
467 int r_samples,
468 int theta_samples)
469 {
470 std::vector<std::complex<double>> z_grid;
471 z_grid.reserve(r_samples * theta_samples);
472
473 for (int i = 0; i < r_samples; ++i) {
474 double r = r_min + i * (r_max - r_min) / (r_samples - 1);
475 for (int j = 0; j < theta_samples; ++j) {
476 double theta = 2.0 * M_PI * j / theta_samples;
477 std::complex<double> z = std::polar(r, theta);
